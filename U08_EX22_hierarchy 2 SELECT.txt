SELECT
 	PARENT_ID,
  SUM(AMOUNT) 
FROM PROD_SALES as sales 
		INNER JOIN  
		HIERARCHY( 
		SOURCE ( SELECT
  			PROD_NAME as NODE_ID,
  			PROD_GROUP as PARENT_ID 
  				FROM PROD_HIER ) 
		START WHERE PROD_GROUP = 'COMPUTER' 
 		DEPTH 4 
		ORPHAN IGNORE ) as hier 
 		on sales.PROD_NAME = hier.NODE_ID 
group by PARENT_ID
